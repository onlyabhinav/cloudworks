        class MultiEnvironmentMonitor {
            constructor() {
                this.environments = new Map();
                this.services = new Map();
                this.previousStates = new Map();
                this.monitoringIntervals = new Map();
                this.isGlobalMonitoring = false;
                this.currentEditingEnv = null;
                
                this.initializeElements();
                this.attachEventListeners();
                this.requestNotificationPermission();
                this.loadEnvironments();
            }

            initializeElements() {
                // Main controls
                this.globalRefreshIntervalSelect = document.getElementById('globalRefreshInterval');
                this.startAllButton = document.getElementById('startAllMonitoring');
                this.stopAllButton = document.getElementById('stopAllMonitoring');
                this.addEnvironmentButton = document.getElementById('addEnvironment');

                // Stats
                this.totalEnvironmentsElement = document.getElementById('totalEnvironments');
                this.activeEnvironmentsElement = document.getElementById('activeEnvironments');
                this.totalServicesElement = document.getElementById('totalServices');
                this.totalUpServicesElement = document.getElementById('totalUpServices');
                this.totalDownServicesElement = document.getElementById('totalDownServices');
                this.globalUptimeElement = document.getElementById('globalUptime');

                // Containers
                this.environmentsGrid = document.getElementById('environmentsGrid');
                this.serviceContainer = document.getElementById('serviceContainer');
                this.environmentFilter = document.getElementById('environmentFilter');
                this.lastUpdated = document.getElementById('lastUpdated');

                // Modal
                this.modal = document.getElementById('environmentModal');
                this.modalTitle = document.getElementById('modalTitle');
                this.closeModal = document.querySelector('.close');
                this.cancelModal = document.getElementById('cancelModal');
                this.saveEnvironment = document.getElementById('saveEnvironment');
                this.testConnectionModal = document.getElementById('testConnectionModal');

                // Modal form elements
                this.envName = document.getElementById('envName');
                this.envDescription = document.getElementById('envDescription');
                this.envEurekaUrl = document.getElementById('envEurekaUrl');
                this.envColor = document.getElementById('envColor');

                // Auth elements
                this.authTabs = document.querySelectorAll('.auth-tab');
                this.authContents = document.querySelectorAll('.auth-content');
                this.modalBasicUsername = document.getElementById('modalBasicUsername');
                this.modalBasicPassword = document.getElementById('modalBasicPassword');
                this.modalBearerToken = document.getElementById('modalBearerToken');
                this.modalApiKeyName = document.getElementById('modalApiKeyName');
                this.modalApiKeyValue = document.getElementById('modalApiKeyValue');
                this.modalCustomHeaders = document.getElementById('modalCustomHeaders');
                this.connectionTestResult = document.getElementById('connectionTestResult');
            }

            attachEventListeners() {
                this.startAllButton.addEventListener('click', () => this.startAllMonitoring());
                this.stopAllButton.addEventListener('click', () => this.stopAllMonitoring());
                this.addEnvironmentButton.addEventListener('click', () => this.openEnvironmentModal());
                
                this.environmentFilter.addEventListener('change', () => this.filterServices());

                // Modal events
                this.closeModal.addEventListener('click', () => this.closeEnvironmentModal());
                this.cancelModal.addEventListener('click', () => this.closeEnvironmentModal());
                this.saveEnvironment.addEventListener('click', () => this.saveEnvironmentConfig());
                this.testConnectionModal.addEventListener('click', () => this.testEnvironmentConnection());

                // Auth tab switching
                this.authTabs.forEach(tab => {
                    tab.addEventListener('click', (e) => {
                        e.preventDefault();
                        this.switchAuthTab(e.target.dataset.tab);
                    });
                });

                // Close modal when clicking outside
                window.addEventListener('click', (e) => {
                    if (e.target === this.modal) {
                        this.closeEnvironmentModal();
                    }
                });
            }

            async requestNotificationPermission() {
                if ('Notification' in window && Notification.permission === 'default') {
                    await Notification.requestPermission();
                }
            }

            loadEnvironments() {
                // Load from localStorage if available
                const saved = localStorage.getItem('eureka_environments');
                if (saved) {
                    try {
                        const envs = JSON.parse(saved);
                        envs.forEach(env => {
                            this.environments.set(env.id, env);
                        });
                    } catch (error) {
                        console.error('Error loading environments:', error);
                    }
                }
                this.updateEnvironmentsUI();
                this.updateGlobalStats();
            }

            saveEnvironments() {
                const envArray = Array.from(this.environments.values());
                localStorage.setItem('eureka_environments', JSON.stringify(envArray));
            }

            openEnvironmentModal(environmentId = null) {
                this.currentEditingEnv = environmentId;
                
                if (environmentId) {
                    const env = this.environments.get(environmentId);
                    this.modalTitle.textContent = 'Edit Environment';
                    this.populateModalForm(env);
                } else {
                    this.modalTitle.textContent = 'Add New Environment';
                    this.clearModalForm();
                }
                
                this.modal.style.display = 'block';
            }

            closeEnvironmentModal() {
                this.modal.style.display = 'none';
                this.currentEditingEnv = null;
                this.clearModalForm();

            populateModalForm(env) {
                this.envName.value = env.name;
                this.envDescription.value = env.description || '';
                this.envEurekaUrl.value = env.eurekaUrl;
                this.envColor.value = env.color || '#28a745';

                // Populate auth fields based on auth type
                if (env.auth) {
                    this.switchAuthTab(env.auth.type);
                    
                    switch (env.auth.type) {
                        case 'basic':
                            this.modalBasicUsername.value = env.auth.username || '';
                            this.modalBasicPassword.value = env.auth.password || '';
                            break;
                        case 'bearer':
                            this.modalBearerToken.value = env.auth.token || '';
                            break;
                        case 'apikey':
                            this.modalApiKeyName.value = env.auth.keyName || 'X-API-Key';
                            this.modalApiKeyValue.value = env.auth.keyValue || '';
                            break;
                        case 'custom':
                            this.modalCustomHeaders.value = JSON.stringify(env.auth.headers || {}, null, 2);
                            break;

            clearModalForm() {

            switchAuthTab(tabName) {

            async testEnvironmentConnection() {
                const originalText = this.testConnectionModal.textContent;
                this.testConnectionModal.textContent = 'Testing...';
                this.testConnectionModal.disabled = true;
                this.connectionTestResult.innerHTML = '';

                try {
                    const eurekaUrl = this.envEurekaUrl.value.trim();
                    if (!eurekaUrl) {
                        throw new Error('Please enter Eureka server URL');
                    }

                    const headers = this.buildAuthHeadersFromModal();
                    const response = await fetch(`${eurekaUrl}/eureka/apps`, { headers });

                    if (!response.ok) {
                        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
                    }

                    this.connectionTestResult.innerHTML = '<div class="connection-status success">✅ Connection successful</div>';
                } catch (error) {
                    this.connectionTestResult.innerHTML = `<div class="connection-status error">❌ Connection failed: ${error.message}</div>`;
                } finally {
                    this.testConnectionModal.textContent = originalText;
                    this.testConnectionModal.disabled = false;
                }
            }

            buildAuthHeadersFromModal() {
                const headers = {
                    'Accept': 'application/json',
                    'Content-Type': 'application/json'
                };

                const activeTab = document.querySelector('.auth-tab.active').dataset.tab;

                switch (activeTab) {
                    case 'basic':
                        if (this.modalBasicUsername.value && this.modalBasicPassword.value) {
                            const credentials = btoa(`${this.modalBasicUsername.value}:${this.modalBasicPassword.value}`);
                            headers['Authorization'] = `Basic ${credentials}`;
                        }
                        break;
                    case 'bearer':
                        if (this.modalBearerToken.value) {
                            headers['Authorization'] = `Bearer ${this.modalBearerToken.value}`;
                        }
                        break;
                    case 'apikey':
                        if (this.modalApiKeyName.value && this.modalApiKeyValue.value) {
                            headers[this.modalApiKeyName.value] = this.modalApiKeyValue.value;
                        }
                        break;
                    case 'custom':
                        if (this.modalCustomHeaders.value) {
                            try {
                                const customHeadersObj = JSON.parse(this.modalCustomHeaders.value);
                                Object.assign(headers, customHeadersObj);
                            } catch (error) {
                                console.warn('Invalid JSON in custom headers:', error);
                            }
                        }
                        break;
                }

                return headers;
            }

            saveEnvironmentConfig() {
                const name = this.envName.value.trim();
                const eurekaUrl = this.envEurekaUrl.value.trim();

                if (!name || !eurekaUrl) {
                    this.showNotification('Please fill in required fields (Name and Eureka URL)', 'error');
                    return;
                }

                const envId = this.currentEditingEnv || this.generateId();
                const activeTab = document.querySelector('.auth-tab.active').dataset.tab;
                
                let auth = null;
                if (activeTab !== 'none') {
                    auth = { type: activeTab };
                    
                    switch (activeTab) {
                        case 'basic':
                            auth.username = this.modalBasicUsername.value;
                            auth.password = this.modalBasicPassword.value;
                            break;
                        case 'bearer':
                            auth.token = this.modalBearerToken.value;
                            break;
                        case 'apikey':
                            auth.keyName = this.modalApiKeyName.value;
                            auth.keyValue = this.modalApiKeyValue.value;
                            break;
                        case 'custom':
                            try {
                                auth.headers = JSON.parse(this.modalCustomHeaders.value || '{}');
                            } catch (error) {
                                this.showNotification('Invalid JSON in custom headers', 'error');
                                return;
                            }
                            break;
                    }
                }

                const environment = {
                    id: envId,
                    name: name,
                    description: this.envDescription.value.trim(),
                    eurekaUrl: eurekaUrl,
                    color: this.envColor.value,
                    auth: auth,
                    isMonitoring: false,
                    lastUpdated: null,
                    stats: {
                        total: 0,
                        up: 0,
                        down: 0
                    }
                };

                this.environments.set(envId, environment);
                this.saveEnvironments();
                this.updateEnvironmentsUI();
                this.updateEnvironmentFilter();
                this.updateGlobalStats();
                this.closeEnvironmentModal();
                
                const action = this.currentEditingEnv ? 'updated' : 'added';
                this.showNotification(`Environment ${action} successfully`, 'success');
            }

            startMonitoring(envId) {
                const env = this.environments.get(envId);
                if (!env || env.isMonitoring) return;

                env.isMonitoring = true;
                this.fetchServices(envId);
                
                const interval = parseInt(this.globalRefreshIntervalSelect.value) * 1000;
                const intervalId = setInterval(() => this.fetchServices(envId), interval);
                this.monitoringIntervals.set(envId, intervalId);

                this.updateEnvironmentsUI();
                this.updateGlobalStats();
            }

            stopMonitoring(envId) {
                const env = this.environments.get(envId);
                if (!env || !env.isMonitoring) return;

                env.isMonitoring = false;
                
                const intervalId = this.monitoringIntervals.get(envId);
                if (intervalId) {
                    clearInterval(intervalId);
                    this.monitoringIntervals.delete(envId);
                }

                this.updateEnvironmentsUI();
                this.updateGlobalStats();
            }

            async fetchServices(envId) {
                const env = this.environments.get(envId);
                if (!env) return;

                try {
                    const headers = this.buildAuthHeaders(env);
                    const response = await fetch(`${env.eurekaUrl}/eureka/apps`, { headers });

                    if (!response.ok) {
                        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
                    }

                    const data = await response.json();
                    this.processEurekaData(data, envId);
                    env.lastUpdated = Date.now();
                    this.updateEnvironmentsUI();
                    this.updateGlobalStats();
                    this.filterServices();
                    this.lastUpdated.textContent = `Last updated: ${new Date().toLocaleString()}`;
                    
                } catch (error) {
                    console.error(`Error fetching services for ${env.name}:`, error);
                    this.showNotification(`Failed to fetch services from ${env.name}: ${error.message}`, 'error');
                }
            }

            buildAuthHeaders(env) {
                const headers = {
                    'Accept': 'application/json',
                    'Content-Type': 'application/json'
                };

                if (!env.auth) return headers;

                switch (env.auth.type) {
                    case 'basic':
                        if (env.auth.username && env.auth.password) {
                            const credentials = btoa(`${env.auth.username}:${env.auth.password}`);
                            headers['Authorization'] = `Basic ${credentials}`;
                        }
                        break;
                    case 'bearer':
                        if (env.auth.token) {
                            headers['Authorization'] = `Bearer ${env.auth.token}`;
                        }
                        break;
                    case 'apikey':
                        if (env.auth.keyName && env.auth.keyValue) {
                            headers[env.auth.keyName] = env.auth.keyValue;
                        }
                        break;
                    case 'custom':
                        if (env.auth.headers) {
                            Object.assign(headers, env.auth.headers);
                        }
                        break;
                }

                return headers;
            }

            processEurekaData(data, envId) {
                const env = this.environments.get(envId);
                let totalInstances = 0;
                let upInstances = 0;

                // Clear existing services for this environment
                this.services.forEach((serviceInfo, serviceKey) => {
                    if (serviceInfo.environmentId === envId) {
                        this.services.delete(serviceKey);
                    }
                });

                if (data.applications && data.applications.application) {
                    const applications = Array.isArray(data.applications.application) 
                        ? data.applications.application 
                        : [data.applications.application];

                    applications.forEach(app => {
                        const instances = Array.isArray(app.instance) ? app.instance : [app.instance];
                        
                        instances.forEach(instance => {
                            const serviceKey = `${envId}-${app.name}-${instance.instanceId}`;
                            const isUp = instance.status === 'UP';
                            
                            const serviceInfo = {
                                environmentId: envId,
                                environmentName: env.name,
                                environmentColor: env.color,
                                name: app.name,
                                instanceId: instance.instanceId,
                                status: instance.status,
                                homePageUrl: instance.homePageUrl,
                                healthCheckUrl: instance.healthCheckUrl,
                                statusPageUrl: instance.statusPageUrl,
                                hostName: instance.hostName,
                                ipAddr: instance.ipAddr,
                                port: instance.port?.$,
                                lastUpdatedTimestamp: instance.lastUpdatedTimestamp,
                                lastDirtyTimestamp: instance.lastDirtyTimestamp,
                                isUp: isUp
                            };

                            this.services.set(serviceKey, serviceInfo);
                            totalInstances++;
                            if (isUp) upInstances++;

                            // Check for status changes
                            const previousState = this.previousStates.get(serviceKey);
                            if (previousState && previousState.isUp !== isUp) {
                                this.notifyStatusChange(serviceInfo, isUp);
                            }
                        });
                    });
                }

                // Update environment stats
                env.stats = {
                    total: totalInstances,
                    up: upInstances,
                    down: totalInstances - upInstances
                };

                // Update previous states
                this.services.forEach((serviceInfo, serviceKey) => {
                    if (serviceInfo.environmentId === envId) {
                        this.previousStates.set(serviceKey, serviceInfo);
                    }
                });
            }

            updateGlobalStats() {
                const totalEnvs = this.environments.size;
                const activeEnvs = Array.from(this.environments.values()).filter(env => env.isMonitoring).length;
                
                let totalServices = 0;
                let totalUp = 0;
                let totalDown = 0;

                this.environments.forEach(env => {
                    totalServices += env.stats.total;
                    totalUp += env.stats.up;
                    totalDown += env.stats.down;
                });

                const globalUptime = totalServices > 0 ? Math.round((totalUp / totalServices) * 100) : 0;

                this.totalEnvironmentsElement.textContent = totalEnvs;
                this.activeEnvironmentsElement.textContent = activeEnvs;
                this.totalServicesElement.textContent = totalServices;
                this.totalUpServicesElement.textContent = totalUp;
                this.totalDownServicesElement.textContent = totalDown;
                this.globalUptimeElement.textContent = `${globalUptime}%`;
            }

            }
